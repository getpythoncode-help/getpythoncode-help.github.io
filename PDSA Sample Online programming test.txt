def sumof3squares(n):
  f=0
  for i in range(1,int(n**(0.5)),1):
    for j in range(1,int(n**(0.5)),1):
      for k in range(1,int(n**(0.5)),1):
        if (i*i + j*j + k*k)==n:
          f=1
  return (f==1)

def intersect(l1,l2):
  l1s=set(l1)
  l2s=set(l2)
  ls = l1s.intersection(l2s)
  return list(ls)
