from math import *
def sumofcubes(n):
  s=[]
  c=[]
  for i in range(1,floor(sqrt(n))):
    s.append(i**3)
  for i in range(0,len(s)-1):
    for j in range(i+1,len(s)):
      c.append(s[i]+s[j])
  for i in c:
    if n == i:
      return True
  else:
    return False

def listdup(l):
    con = {}
    tup = list()
    for i in l:
        if i not in con:
            con[i] = 1
        else:
            con[i] = con.get(i, 0) + 1
    for (key, value) in con.items():
        if value > 1:
            tup.append(key)

    return tup
