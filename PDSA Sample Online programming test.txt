static int findMissing(int array[], int s, int e)
    {
        if (s > e)
            return e + 1;
        if (s != array[s])
            return s;
  
        int mid = (s + e) / 2;
 
        
        if (array[mid] == mid)
            return findMissing(array, mid+1, e);
 
        return findMissing(array, s, mid);
    }
