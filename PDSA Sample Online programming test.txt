def perfect(n):
  sum=0
  for i in range(1,n):
    if n%i==0:
      sum+=i
  return(sum==n)

def sublist(l1,l2):
  f=0
  for i in range(len(l2)-len(l1)):
    if l2[i:len(l1)]==l1 and len(l1) != 1:
      f=2
    elif len(l1)==1 and l1[0] in l2:
      f=2
  return(f==2)
