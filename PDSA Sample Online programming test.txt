def sum3cubes(n):
  sum=0
  for i in range(1,n):
    for j in range(1,n):
      for k in range(1,n):
        if (i*i*i+j*j*j+k*k*k) == n:
          sum+=1
  if sum>=1: 
    return True
  else:
    return False
