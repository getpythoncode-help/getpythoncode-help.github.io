def sumofsquares(n) : 
  i = 1
  while i * i <= n : 
    j = 1
    while(j * j <= n) : 
      if (i * i + j * j == n) : 
        return True
      j = j + 1
    i = i + 1  
  return False

def subsequence(l1, l2):
  count = 0
  for i in l1:
    if i in l2:
      l2.remove(i)
      count += 1
  if count == len(l1):
    return(True)
  else:
    return(False)
